Requesting Data

To add Trading Economics data to your algorithm use the AddData() system to request the data. As with all datasets, you should save a reference to your symbol for easy use later in your algorithm. For detailed documentation on using custom data see Importing Custom Data. 

// U.S. Federal Reserve interest rate
var calendarInterest = AddData<TradingEconomicsCalendar>(TradingEconomics.Calendar.UnitedStates.InterestRate).Symbol;
// U.S. Federal Reserve interest rate
var indicatorInterest = AddData<TradingEconomicsIndicator>(TradingEconomics.Indicator.UnitedStates.InterestRate).Symbol;

# Add U.S. Federal Reserve interest rate data.
calendarInterest = self.AddData(TradingEconomicsCalendar, TradingEconomics.Calendar.UnitedStates.InterestRate).Symbol
# Add U.S. Federal Reserve interest rate data.
indicatorInterest = self.AddData(TradingEconomicsIndicator, TradingEconomics.Indicator.UnitedStates.InterestRate).Symbol

Accessing Data

Data can be accessed via the Slice events. Slice delivers the unique events to your algorithm as they happen. We recommend saving the symbol object when you add the data for easy access to slice later. Data is available in the following resolutions: Daily. You can see an example of the slice accessor in the code below. 

using QuantConnect.Data.Custom.TradingEconomics;
namespace QuantConnect.Algorithm.CSharp {
    public class TradingEconomicsAlgorithm : QCAlgorithm {
        public override void Initialize() {
            SetStartDate(2019, 1, 1);
            
            // U.S. Federal Reserve interest rate
            AddData<TradingEconomicsCalendar>(TradingEconomics.Calendar.UnitedStates.InterestRate);
            // U.S. Federal Reserve interest rate
            AddData<TradingEconomicsIndicator>(TradingEconomics.Indicator.UnitedStates.InterestRate);
        }
        
        public override void OnData(Slice data) {
            // Accessing via Slice event
            var calendarInterest = data.Get<TradingEconomicsCalendar>();
            var indicatorInterest = data.Get<TradingEconomicsIndicator>();
            
            foreach (var calendar in calendarInterest.Values)
            {
                Log($"Unique calendar ID used by Trading Economics: {calendar.CalendarId}, Release time and date in UTC: {calendar.EndTime}, Country name: {calendar.Country}, Indicator category name: {calendar.Category}, Specific event name in the calendar: {calendar.Event}, The period for which released data refers to: {calendar.Reference}, Source of data: {calendar.Source}, Latest released value: {calendar.Actual}, Value for the previous period after the revision (if revision is applicable): {calendar.Previous}, Average forecast among a representative group of economists: {calendar.Forecast}, TradingEconomics own projections: {calendar.TradingEconomicsForecast}, 0 indicates that the time of the event is known, 1 indicates that we only know the date of event, the exact time of event is unknown: {calendar.DateSpan}, Importance of a TradingEconomics information: {calendar.Importance}, Time when new data was inserted or changed: {calendar.LastUpdate}, Value reported in the previous period after revision: {calendar.Revised}, Countrys original name: {calendar.OCountry}, Categorys original name: {calendar.OCategory}, Unique ticker used by Trading Economics: {calendar.Ticker}, Unique symbol used by Trading Economics: {calendar.TESymbol}, Indicates whether the Actual, Previous, Forecast, TradingEconomicsForecast fields are reported as percent values: {calendar.IsPercentage}");
            }
            foreach (var indicator in indicatorInterest.Values)
            {
                Log($"Country name: {indicator.Country}, Indicator category name: {indicator.Category}, Release time and date in UTC: {indicator.EndTime}, Value: {indicator.Value}, Frequency of the indicator: {indicator.Frequency}, Time when new data was inserted or changed: {indicator.LastUpdate}, Unique symbol used by Trading Economics: {indicator.HistoricalDataSymbol}");
            }
        }
    }
}
