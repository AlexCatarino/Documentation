Data can be accessed via Slice events. Slice delivers unique events to your algorithm as they happen. We recommend saving the symbol object when you add the data for easy access to slice later. Data is available in daily resolution. You can see an example of the slice accessor in the code below.

class SECReportAlgorithm(QCAlgorithm):
    def OnData(self, data: Slice) -> None:
        if self.report8K in data:
            report8K_data = data.Get(SECReport8K, self.report8K)
            documentsCount = len(report8K_data.Report.Documents);
            self.Log(f'{documentsCount} found for {self.aapl} on {self.Time}');

namespace QuantConnect
{
    public class SECReportAlgorithm : QCAlgorithm
    {
        if (data.ContainsKey(_report8K))
        {
            var report8KData = data.Get(_report8K);
            var documentsCount = report8KData.Report.Documents.Count;
            Log($"{documentsCount} found for {_aapl} on {Time}");
        }
    }
}
