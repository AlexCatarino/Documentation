You can't request historical values for CoarseFundamental members, but you can request historical market data on the security by using the Symbol member of the CoarseFundamental object. If there is no data in the period you request, the history result will be empty. The following example gets the historical data for securities by using the History API.

def CoarseSelectionFunction(self, coarse: List[CoarseFundamental]) -> List[Symbol]:

    selected_symbols = [x.Symbol 
        for x in sorted(coarse, key=lambda x: x.DollarVolume, reverse=True)[:3]]

    history = self.History(selected_symbols, 10, Resolution.Daily)

    return selected_symbols

public IEnumerable<Symbol> CoarseSelectionFunction(IEnumerable<CoarseFundamental> coarse)
{
    var selectedSymbols = coarse
        .OrderByDescending(x => x.DollarVolume)
        .Take(3)
        .Select(x => x.Symbol)
        .ToList();

    var history = History(selectedSymbols, 10, Resolution.Daily);

    return selectedSymbols;
}
