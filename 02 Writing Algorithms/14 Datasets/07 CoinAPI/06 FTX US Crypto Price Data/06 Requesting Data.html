To add FTX US data from CoinAPI to your algorithm, use the AddCrypto() method to request the data. As with all datasets, you should save a reference to your symbol for easy use later in your algorithm. When requesting data, you have the option to use several Crypto markets. For detailed documentation on requesting market data, see Selecting Asset Data.

class FTXUSAlgorithm(QCAlgorithm):

    def Initialize(self):
        self.SetStartDate(2020, 6, 1)
        self.SetEndDate(2021, 6, 1)
        self.SetCash(100000)

        # FTX US accepts both cash and margin account types
        self.SetBrokerageModel(BrokerageName.FTXUS, AccountType.Margin)
        
        self.btcusd = self.AddCrypto("BTCUSD", Resolution.Minute, Market.FTXUS).Symbol
namespace QuantConnect
{
    public class FTXUSAlgorithm : QCAlgorithm
    {
        private Symbol _btcusd;
        
        public override void Initialize()
        {
            SetStartDate(2020, 6, 1);
            SetEndDate(2021, 6, 1);
            SetCash(100000);

            // FTX US accepts both cash and margin account types
            SetBrokerageModel(BrokerageName.FTXUS, AccountType.Margin);
            
            _btcusd = AddCrypto("BTCUSD", Resolution.Minute, Market.FTXUS).Symbol;
        }
    }
}
