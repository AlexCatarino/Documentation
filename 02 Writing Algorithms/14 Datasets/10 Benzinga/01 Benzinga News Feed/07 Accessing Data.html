Data can be accessed via Slice events. Slice delivers unique events to your algorithm as they happen. We recommend saving the symbol object when you add the data for easy access to slice later. Data is available in second resolution. You can see an example of the slice accessor in the code below.

def OnData(self, slice: Slice) -> None:
    if slice.ContainsKey(self.benzinga_symbol):
        article = slice[self.benzinga_symbol]
        self.Log(f"Received Benzinga News article at {self.Time}; ID: {article.Id}; Author {article.Author}; Created at: {article.CreatedAt}; Updated at: {article.UpdatedAt}; Title: {article.Title}; Teaser {article.Teaser}; Contents: {article.Contents}; Symbols: {article.Symbols}; Tags: {article.Tags}; EndTime {article.EndTime}")

public override void OnData(Slice slice)
{
    if (slice.ContainsKey(_benzingaSymbol))
    {
        var article = slice[_benzingaSymbol];
        Log($"Received Benzinga News article at {Time}; ID: {article.Id}; Author {article.Author}; Created at: {article.CreatedAt}; Updated at: {article.UpdatedAt}; Title: {article.Title}; Teaser {article.Teaser}; Contents: {article.Contents}; Symbols: {article.Symbols}; Tags: {article.Tags}; EndTime {article.EndTime}");
    }
}
