The following snippet demonstrates how to request data from the Bitcoin Metadata dataset:

self.btcusd = self.AddCrypto("BTCUSD", Resolution.Daily, Market.Bitfinex).Symbol
bitcoin_metadata_symbol = self.AddData(BitcoinMetadata, self.btcusd).Symbol 
btcusd = AddCrypto("BTCUSD", Resolution.Daily, Market.Bitfinex).Symbol;
bitcoinMetadataSymbol = AddData<BitcoinMetadata>(btcusd).Symbol;
