To add Coinbase Pro data from CoinAPI to your algorithm, use the AddCrypto() method to request the data. As with all datasets, you should save a reference to your symbol for easy use later in your algorithm. When requesting data, you have the option to use several crypto markets. For detailed documentation on requesting market data, see Selecting Asset Data.

class CoinAPIDataAlgorithm(QCAlgorithm):

    def Initialize(self):
        self.SetStartDate(2020, 6, 1)
        self.SetEndDate(2021, 6, 1)
        self.SetCash(100000)

        # Coinbase Pro accepts Cash account type only, AccountType.Margin will result in an exception.
        self.SetBrokerageModel(BrokerageName.GDAX, AccountType.Cash)
        
        self.btcusd = self.AddCrypto("BTCUSD", Resolution.Minute, Market.GDAX).Symbol
namespace QuantConnect
{
    public class CoinAPIDataAlgorithm : QCAlgorithm
    {
        private Symbol _btcusd;
        
        public override void Initialize()
        {
            SetStartDate(2020, 6, 1);
            SetEndDate(2021, 6, 1);
            SetCash(100000);
        
            // Coinbase Pro accepts Cash account type only, AccountType.Margin will result in an exception.
            SetBrokerageModel(BrokerageName.GDAX, AccountType.Cash);
            
            _btcusd = AddCrypto("BTCUSD", Resolution.Minute, Market.GDAX).Symbol;
        }
    }
}
