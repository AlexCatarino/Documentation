To add a Universe Selection model, in the Initialize method, call the AddUniverseSelection method. The resolution of the assets in the universe is set by the UniverseSettings.

UniverseSettings.Resolution = Resolution.Minute;
var symbols = new [] {QuantConnect.Symbol.Create("SPY", SecurityType.Equity, Market.USA)};
AddUniverseSelection(new ManualUniverseSelectionModel(symbols));

self.UniverseSettings.Resolution = Resolution.Minute
symbols = [Symbol.Create("SPY", SecurityType.Equity, Market.USA)]
self.AddUniverseSelection(ManualUniverseSelectionModel(symbols))
