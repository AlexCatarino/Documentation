Custom indicators subsclass the PythonIndicator class. The indicator must have Name attribute, Time attribute, Value attribute, and an Update method. The Update method should accept an IndicatorDataPoint, QuoteBar, or TradeBar and return a boolean that represents if the indicator is ready.

Custom indicators subsclass the IndicatorBase<IndicatorDataPoint> class.

The following definition provides an example of a custom simple moving average indicator:

public class CustomSimpleMovingAverage : IndicatorBase<IndicatorDataPoint> 
{
    public Sum RollingSum;

    public override bool IsReady => RollingSum.IsReady;

    public new long Samples = 0;

    public CustomSimpleMovingAverage(string name, int period) : base(name)
    {
        RollingSum = new Sum(name + "_Sum", period);
    }

    protected override decimal ComputeNextValue(IndicatorDataPoint input)
    {
        return default(decimal);
    }
    
    public override bool Update(IBaseData input)
    {
        Samples++;
        RollingSum.Update(input.Time, input.Value);
        Current = new IndicatorDataPoint(input.Time, RollingSum.Current.Value / RollingSum.WarmUpPeriod);
        return IsReady;
    }
    
    public override void Reset()
    {
        Samples = 0;
        RollingSum.Reset();
        Current = new IndicatorDataPoint(DateTime.MinValue, default(decimal));
    }
}
class CustomSimpleMovingAverage(PythonIndicator):
    def __init__(self, name, period):
        super().__init__()
        self.Name = name
        self.Time = datetime.min
        self.Value = 0
        self.queue = deque(maxlen=period)

    def Update(self, input):
        self.queue.appendleft(input.Value)
        count = len(self.queue)
        self.Time = input.Time
        self.Value = sum(self.queue) / count
        return count == self.queue.maxlen
