The .Minus(IIndicator) extension subtracts the .Current.Value of two indicators or subtracts a fixed value from the .Current.Value of an indicator.

// Subtract the output of two indicators
var smaShort = SMA("SPY", 14);
var smaLong = SMA("SPY", 21);
var smaDifference = smaShort.Minus(smaLong);

// Sum the output of an indicator and a fixed value
var smaMinusValue = smaShort.Minus(10);
# Sum the output of two indicators
sma_short = self.SMA("SPY", 14)
sma_long = self.SMA("SPY", 21)
sma_difference = sma_short.Minus(sma_long)

# Sum the output of an indicator and a fixed value
sma_minus_value = sma_short.Minus(10)

If you pass an indicator to the .Minus(IIndicator) extension, you can name the composite indicator.

var namedIndicator = smaShort.Minus(smaLong, "SMA Difference");
named_indicator = sma_short.Minus(sma_long, "SMA Difference")
