We provide a helper method which aims to make plotting indicators simple. For further information on the charting API please see our Charting section.

private Stochastic _sto;
// In Initialize()
_sto = STO(symbol, period, resolution);

// In OnData()
if (_sto.IsReady)
{
    Plot("My Indicators", "STO Fast Stoch", _sto.FastStoch);
    Plot("My Indicators", "STO Stoch K", _sto.StochK);
    Plot("My Indicators", "STO Stoch D", _sto.StochD);
    Plot("My Indicators", "STO Is Ready", _sto.IsReady);
    Plot("My Indicators", "STO Warm Up Period", _sto.WarmUpPeriod);
    Plot("My Indicators", "STO Current", _sto.Current);
    Plot("My Indicators", "STO Name", _sto.Name);
    Plot("My Indicators", "STO Samples", _sto.Samples);
}
