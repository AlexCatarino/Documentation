We provide a helper method which aims to make plotting indicators simple. For further information on the charting API please see our Charting section.

private Maximum _max;
// In Initialize()
_max = MAX(symbol, period, resolution, selector);

// In OnData()
if (_max.IsReady)
{
    Plot("My Indicators", "MAX Periods Since Maximum", _max.PeriodsSinceMaximum);
    Plot("My Indicators", "MAX Is Ready", _max.IsReady);
    Plot("My Indicators", "MAX Warm Up Period", _max.WarmUpPeriod);
    Plot("My Indicators", "MAX Period", _max.Period);
    Plot("My Indicators", "MAX Current", _max.Current);
    Plot("My Indicators", "MAX Name", _max.Name);
    Plot("My Indicators", "MAX Samples", _max.Samples);
}
