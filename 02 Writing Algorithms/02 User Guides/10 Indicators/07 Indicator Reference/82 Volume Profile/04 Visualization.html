We provide a helper method which aims to make plotting indicators simple. For further information on the charting API please see our Charting section.

private VolumeProfile _vp;
// In Initialize()
_vp = VP(symbol, period, valueAreaVolumePercentage, priceRangeRoundOff, resolution, selector);

// In OnData()
if (_vp.IsReady)
{
    Plot("My Indicators", "VP Volume Per Price", _vp.VolumePerPrice);
    Plot("My Indicators", "VP Profile High", _vp.ProfileHigh);
    Plot("My Indicators", "VP Profile Low", _vp.ProfileLow);
