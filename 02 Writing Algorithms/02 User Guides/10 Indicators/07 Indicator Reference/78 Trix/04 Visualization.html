We provide a helper method which aims to make plotting indicators simple. For further information on the charting API please see our Charting section.

private Trix _trix;
// In Initialize()
_trix = TRIX(symbol, period, resolution, selector);

// In OnData()
if (_trix.IsReady)
{
    Plot("My Indicators", "TRIX Is Ready", _trix.IsReady);
    Plot("My Indicators", "TRIX Warm Up Period", _trix.WarmUpPeriod);
    Plot("My Indicators", "TRIX Current", _trix.Current);
    Plot("My Indicators", "TRIX Name", _trix.Name);
    Plot("My Indicators", "TRIX Samples", _trix.Samples);
}
