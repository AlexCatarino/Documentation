We provide a helper method which aims to make plotting indicators simple. For further information on the charting API please see our Charting section.

_cci = CCI(symbol, period, movingAverageType, resolution, selector);

// In OnData()
if (_cci.IsReady)
{
    Plot("My Indicators", "CCI Moving Average Type", _cci.MovingAverageType);
    Plot("My Indicators", "CCI Typical Price Average", _cci.TypicalPriceAverage);
    Plot("My Indicators", "CCI Typical Price Mean Deviation", _cci.TypicalPriceMeanDeviation);
    Plot("My Indicators", "CCI Is Ready", _cci.IsReady);
    Plot("My Indicators", "CCI Warm Up Period", _cci.WarmUpPeriod);
    Plot("My Indicators", "CCI Current", _cci.Current);

}
