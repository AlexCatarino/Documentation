// In OnData()
if (_sr.IsReady)

    Plot("My Indicators", "SR Warm Up Period", _sr.WarmUpPeriod);
    Plot("My Indicators", "SR Is Ready", _sr.IsReady);
    Plot("My Indicators", "SR Current", _sr.Current);
