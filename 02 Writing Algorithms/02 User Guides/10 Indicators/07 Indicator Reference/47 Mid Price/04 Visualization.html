We provide a helper method which aims to make plotting indicators simple. For further information on the charting API please see our Charting section.

private MidPrice _midprice;
// In Initialize()
_midprice = MIDPRICE(symbol, period, resolution, selector);

// In OnData()
if (_midprice.IsReady)
{
    Plot("My Indicators", "MIDPRICE Is Ready", _midprice.IsReady);
    Plot("My Indicators", "MIDPRICE Warm Up Period", _midprice.WarmUpPeriod);
    Plot("My Indicators", "MIDPRICE Current", _midprice.Current);
    Plot("My Indicators", "MIDPRICE Name", _midprice.Name);
    Plot("My Indicators", "MIDPRICE Samples", _midprice.Samples);
}
