We provide a helper method which aims to make plotting indicators simple. For further information on the charting API please see our Charting section.

private MoneyFlowIndex _mfi;
// In Initialize()
_mfi = MFI(symbol, period, resolution, selector);

// In OnData()
if (_mfi.IsReady)
{
    Plot("My Indicators", "MFI Positive Money Flow", _mfi.PositiveMoneyFlow);
    Plot("My Indicators", "MFI Negative Money Flow", _mfi.NegativeMoneyFlow);
    Plot("My Indicators", "MFI Previous Typical Price", _mfi.PreviousTypicalPrice);
    Plot("My Indicators", "MFI Is Ready", _mfi.IsReady);
    Plot("My Indicators", "MFI Warm Up Period", _mfi.WarmUpPeriod);
    Plot("My Indicators", "MFI Current", _mfi.Current);
    Plot("My Indicators", "MFI Name", _mfi.Name);
    Plot("My Indicators", "MFI Samples", _mfi.Samples);
}
