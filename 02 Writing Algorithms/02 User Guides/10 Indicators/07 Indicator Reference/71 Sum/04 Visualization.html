We provide a helper method which aims to make plotting indicators simple. For further information on the charting API please see our Charting section.

private Sum _sum;
// In Initialize()
_sum = SUM(symbol, period, resolution, selector);

// In OnData()
if (_sum.IsReady)
{
    Plot("My Indicators", "SUM Warm Up Period", _sum.WarmUpPeriod);
    Plot("My Indicators", "SUM Period", _sum.Period);
    Plot("My Indicators", "SUM Is Ready", _sum.IsReady);
    Plot("My Indicators", "SUM Current", _sum.Current);
    Plot("My Indicators", "SUM Name", _sum.Name);
    Plot("My Indicators", "SUM Samples", _sum.Samples);
}
