_trin = TRIN(symbols, resolution);

// In OnData()
if (_trin.IsReady)
{
    Plot("My Indicators", "TRIN A D Ratio", _trin.ADRatio);
    Plot("My Indicators", "TRIN A D V Ratio", _trin.ADVRatio);
    Plot("My Indicators", "TRIN Is Ready", _trin.IsReady);
    Plot("My Indicators", "TRIN Warm Up Period", _trin.WarmUpPeriod);
    Plot("My Indicators", "TRIN Current", _trin.Current);
    Plot("My Indicators", "TRIN Name", _trin.Name);
