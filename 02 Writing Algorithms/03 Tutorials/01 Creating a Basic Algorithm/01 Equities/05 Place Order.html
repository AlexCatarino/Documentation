You must set up the algorithm to check the portfolio state before you can place orders.

To place orders to buy and hold, when the portfolio is not invested, call the SetHoldings method with the Equity Symbol and a portfolio weight.

if (!Portfolio.Invested)
{
    foreach (var symbol in new [] {_spy, _msft, _googl})
    {
        if (data.ContainsKey(symbol) && data[symbol] != null)
        {
            SetHoldings(symbol, 0.33);
        }
    }
}
if not self.Portfolio.Invested:
    for symbols in [self.spy, self.msft, self.googl]:
        if data.ContainsKey(symbol) and data[symbol] is not None:
            self.SetHoldings(symbol, 0.33);
