To set the leverage for all securities in your algorithm, set the UniverseSettings.Leverage property. 
// In Initialize
UniverseSettings.Leverage = 2;
# In Initialize
self.UniverseSettings.Leverage = 2
To set the leverage on a security basis, use a security initializer. 
// In Initialize
SetSecurityInitializer(CustomSecurityInitializer);

private void CustomSecurityInitializer(Security security)
{
    security.SetLeverage(2);
}

# In Initialize
self.SetSecurityInitializer(self.CustomSecurityInitializer)

def CustomSecurityInitializer(self, security):
    security.SetLeverage(2)
