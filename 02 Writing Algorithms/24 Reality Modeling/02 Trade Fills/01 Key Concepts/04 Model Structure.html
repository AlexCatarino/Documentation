Fill Models should extend the FillModel class. To implement your own fill model, override the methods in the FillModel class you wish to change. The class has a dedicated method for each order type. The method receives a Security and Order object and returns an OrderEvent object that contains information about the order status, fill quantity, and errors. 

public class MyFillModel : FillModel {

    public override OrderEvent MarketFill(Security asset, MarketOrder order) {
        return base.MarketFill(asset, order);
    }

    public override OrderEvent LimitFill(Security asset, MarketOrder order) {
        return base.LimitFill(asset, order);
    }

    public override OrderEvent LimitIfTouchedFill(Security asset, MarketOrder order) {
        return base.LimitIfTouchedFill(asset, order);
    }

    public override OrderEvent StopMarketFill(Security asset, MarketOrder order) {
        return base.StopMarketFill(asset, order);
    }

    public override OrderEvent StopLimitFill(Security asset, MarketOrder order) {
        return base.StopLimitFill(asset, order);
    }

    public override OrderEvent MarketOnOpenFill(Security asset, MarketOrder order) {
        return base.MarketOnOpenFill(asset, order);
    }

    public override OrderEvent MarketOnCloseFill(Security asset, MarketOrder order) {
        return base.MarketOnCloseFill(asset, order);
    }
}

class MyFillModel(FillModel):

    def MarketFill(self, asset, order):
        return super().MarketFill(asset, order)

    def LimitFill(self, asset, order):
        return super().LimitFill(asset, order)

    def LimitIfTouchedFill(self, asset, order):
        return super().LimitIfTouchedFill(asset, order)

    def StopMarketFill(self, asset, order):
        return super().StopMarketFill(asset, order)

    def StopLimitFill(self, asset, order):
        return super().StopLimitFill(asset, order)

    def MarketOnOpenFill(self, asset, order):
        return super().MarketOnOpenFill(asset, order)

    def MarketOnCloseFill(self, asset, order):
        return super().MarketOnCloseFill(asset, order)
