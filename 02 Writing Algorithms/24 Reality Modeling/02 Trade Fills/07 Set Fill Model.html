To set the fill model of a security, call the SetFillModel method on the Security object.
// In Initialize
var security = AddEquity("SPY");
security.SetFillModel(new EquityFillModel());
# In Initialize
security = self.AddEquity("SPY")
security.SetFillModel(EquityFillModel())
Alternatively, set the fill model in a security initializer. 
// In Initialize
SetSecurityInitializer(CustomSecurityInitializer);

private void CustomSecurityInitializer(Security security)
{
    security.SetFillModel(new EquityFillModel());
}

# In Initialize
self.SetSecurityInitializer(self.CustomSecurityInitializer)

def CustomSecurityInitializer(self, security):
    security.SetFillModel(EquityFillModel())
