def CustomPeriod(self, dt):
    period = timedelta(7)
    dt = dt.replace(hour=17, minute=0, second=0, microsecond=0)
    delta = 1+dt.weekday()
    if delta > 6:
        delta = 0
    start = dt-timedelta(delta)
    return CalendarInfo(start, period)
