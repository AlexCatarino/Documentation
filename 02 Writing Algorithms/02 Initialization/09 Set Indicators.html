In some cases, you may need to create and warm up indicators in the Initialize() method. 

private Symbol _symbol;
private SimpleMovingAverage _sma;

_symbol = AddEquity("SPY");
_sma = SMA(_symbol, 20);
WarmUpIndicator(_symbol, _sma);
self.symbol = self.AddEquity("SPY")
self.sma = self.SMA(self.symbol, 20)
self.WarmUpIndicator(self.symbol, self.sma)
