The ObjectStore is worked in a format of a hash table. Thus, to access its stored data, we need a key to retrieve the data correspond to that particular key. We can iterate all objects in ObjectStore and get a list of all the keys.

    4. public static List<string> GetObjectStoreKeys(){
    // Instantiate a QuantBook instance.
    var qb = new QuantBook();

    // Initialize a list to hold the keys.
    var keys = new List<string>();
    
    // Call the ObjectStore class.
    var objectStore = qb.ObjectStore;
    
    // Iterate the ObjectStore objects.
    foreach(var kvp in objectStore){
        // Add the key into the list.
        keys.Add(kvp.Key);
    }
    
    return keys;
}
    def get_ObjectStore_keys():
    # Initialize a list to hold the keys.
    keys = []
    
    # Call the ObjectStore class.
    qb = QuantBook()
    object_store = qb.ObjectStore
    
    # Iterate the ObjectStore objects.
    for item in object_store.GetEnumerator():
        # Split the item characters and get the key only.
        key = str(item).split(",")[0][1:]
        
        # Append the key into the list.
        keys.append(key)
        
    return keys
