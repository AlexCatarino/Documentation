We would show an example on using a Datapoint Indicator, a Bar Indicator, a TradeBar Indicator and a Combined Indicator.

// Define a datapoint indicator.
// In this example, we'll be using a 20-period 2-standard-deviation BollingerBands indicator.
var bb = new BollingerBands(20, 2);

// Define a bar indicator (utilizing OHLC information).
// In this example, we'll be using a 10-period-bar AverageTrueRange indicator.
var atr = new AverageTrueRange(10);

// Define a trade bar indicator (utilizing OHLCV information).
// In this example, we'll be using a 20-period-bar VolumeWeightedAveragePriceIndicator indicator.
var vwap = new VolumeWeightedAveragePriceIndicator(20);

// Define a combined indicator.
// In this example, we'll be using a 10-day SimpleMovingAverage of a 10-period RelativeStrengthIndex indicator.
var rsi = new RelativeStrengthIndex(10);
var sma = new SimpleMovingAverage(10);
var smaOfRsi = IndicatorExtensions.Of(sma, rsi);
