The following examples demonstrate some common liquidity universes for US Equities.

Example 1: Large Cap Liquid Universe

The following example selects 20 of the most traded large-cap stocks:

public class LargeCapLiquidUniverseAlgorithm : QCAlgorithm
{
    public override void Initialize()
    {
        AddUniverse(
            fundamental => fundamental
                .Where(f => f.MarketCap >= 1e10m)
                .OrderByDescending(f => f.DollarVolume)
                .Take(20)
                .Select(f => f.Symbol)
        );
    }
}
class LargeCapLiquidUniverseAlgorithm(QCAlgorithm):

    def initialize(self) -> None:
        self.add_universe(self._select_assets)

    def _select_assets(self, fundamental: List[Fundamental]) -> List[Symbol]:
        # Select the most liquid large-cap stocks.
        selected = sorted(
            [f for f in fundamental if f.market_cap >= 1e10], 
            key=lambda f: f.dollar_volume
        )[-20:]
        # Return the symbols of the selected stocks.
        return [f.symbol for f in selected]

Example 2: Illiquid Penny Stocks Universe

Some market makers aim at illiquid penny stocks to earn higher spreads with less captital. The following example selects the most illiquid stocks below $3. 

public class IliquidPennyStockUniverseAlgorithm : QCAlgorithm
{
    public override void Initialize()
    {
        // Select the most illiquid stocks below $3.
        AddUniverse(
            fundamental => fundamental
                .Where(f => f.Price <= 3)
                .OrderBy(f => f.DollarVolume)
                .Take(20)
                .Select(f => f.Symbol)
        );
    }
}
class IliquidPennyStockUniverseAlgorithm(QCAlgorithm):

    def initialize(self):
        self.add_universe(self._select_assets)

    def _select_assets(self, fundamental: List[Fundamental]) -> List[Symbol]:
        # Select the most illiquid stocks below $3.
        selected = sorted(
            [f for f in fundamental if f.price <= 3],
            key=lambda f: f.dollar_volume
        )[:20]
        return [f.symbol for f in selected]
