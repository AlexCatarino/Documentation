Example 1: Large Cap Liquid Universe

Select 20 most traded stocks from large-cap stocks.

public override void Initialize()
{
    // Filter for large-cap stocks (market_cap>=$10 billion), sort by dollar volume, return the top 20 symbols. 
    AddUniverse((fundamental) => fundamental.Where(f => f.HasFundamentalData && f.MarketCap >= 1e10m)
        .OrderedByDescending(f => f.DollarVolume)
        .Take(20)
        .Select(f => f.Symbol));
}
def initialize(self):
    # Initialize algo with custom universe
    self.add_universe(self._selection)

def _selection(self, fundamental: List[Fundamental]) -> List[Symbol]:
    # Filter for large-cap stocks (market_cap >= $10 billion). 
    large_cap_stocks = [f for f in fundamental if f.has_fundamental_data and f.market_cap >= 1e10]
    # Sort the filtered large-cap stocks by dollar volume in decending order.
    sorted_by_dollar_volume = sorted(large_cap_stocks, key=lambda f: f.dollar_volume, reverse=True)
    # Return the symbols of the top 20 stocks by dollar volume. 
    return [f.symbol for f in sorted_by_dollar_volume[:20]]

Example 2: Illiquid Penny Stocks Universe

Some market makers will aim at illiquid penny stocks to earn higher spreads with less captital.

public override void Initialize()
{
    // Filter stocks for (price <= $3), sort by lowest dollar volume, and return the symbols of the lowest 20 (by dollar volume). 
    AddUniverse((fundamental) => fundamental.Where(f => f.Price <= 3)
        .OrderedBy(f => f.DollarVolume)
        .Take(20)
        .Select(f => f.Symbol));
}
 
def initialize(self):
    # Initialize algo with custom universe
    self.add_universe(self._selection)

def _selection(self, fundamental: List[Fundamental]) -> List[Symbol]:
    # Filter stocks for (price <= $3).
    penny_stocks = [f for f in fundamental if f.price <= 3]
    # Sort stocks by dollar volume in ascending order.
    sorted_by_dollar_volume = sorted(penny_stocks, key=lambda f: f.dollar_volume)
    # Return 20 symbols of the penny stocks that have the lowest dollar volume. 
    return [f.symbol for f in sorted_by_dollar_volume[:20]]
