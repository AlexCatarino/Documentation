The following examples demonstrate some common practices for requesting individual Future contract data.

Example 1: Extended Hours Continuous Future Contract

The following example demonstrates how to request data subscription of E-Mini S&P 500 Future contract with the closest expiration. To also subscribe to extended market hours data, set the extendedMarketHoursextended_market_hours argument to trueTrue.

public FutureExampleAlgorithm : QCAlgorithm
{
    private Symbol _future;

    public override void Initialize()
    {
        _future = AddFuture(Futures.Indices.SP500EMini).Symbol;
        // Request all tradable future contracts for filtering.
        var contractSymbols = FutureChainProvider.GetFutureContractList(_future, Time);
        // Obtain the contract with the closest expiry.
        var contractSymbol = contractSymbols.MinBy(x => x.ID.Date);
        // Extended market hours trading allows flexibility and hedging during equity market close.
        _contractSymbol = AddFutureContract(
            contractSymbol,
            extendedMarketHours: true
        );
    }
}
class FutureExampleAlgorithm(QCAlgorithm):
    def initialize(self):
        self._future = self.add_future(Futures.Indices.SP_500_E_MINI).symbol
        # Request all tradable future contracts for filtering.
        contract_symbols = self.future_chain_provider.get_future_contract_list(self._continuous, self.time)
        # Obtain the contract with the closest expiry.
        contract_symbol = sorted(contract_symbols, key=lambda x: x.id.date)[0]
        # Extended market hours trading allows flexibility and hedging during equity market close.
        self._contract_symbol = self.add_future_contract(
            contract_symbol,
            extended_market_hours=True
        ).symbol
