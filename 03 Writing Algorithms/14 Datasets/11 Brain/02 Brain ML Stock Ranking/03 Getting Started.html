The following snippet demonstrates how to request data from the Brain ML Stock Ranking dataset:

from QuantConnect.DataSource import *

self.symbol = self.AddEquity("AAPL", Resolution.Daily).Symbol
self.dataset_symbol = self.AddData(BrainStockRanking2Day, self.symbol).Symbol

self.AddUniverse(BrainStockRankingUniverse, "BrainStockRankingUniverse", Resolution.Daily, self.UniverseSelectionMethod)
using QuantConnect.DataSource;

_symbol = AddEquity("AAPL", Resolution.Daily).Symbol;
_datasetSymbol = AddData<BrainStockRanking2Day>(_symbol).Symbol;

AddUniverse<BrainStockRankingUniverse>("BrainStockRankingUniverse", Resolution.Daily, UniverseSelectionMethod);
