The following snippet demonstrates how to request data from the Bybit Crypto Price dataset:

from QuantConnect.DataSource import *
from QuantConnect.Data.UniverseSelection import *

# Bybit accepts both Cash and Margin account types only.
self.SetBrokerageModel(BrokerageName.Bybit, AccountType.Cash)
self.SetBrokerageModel(BrokerageName.Bybit, AccountType.Margin)

self.symbol = self.AddCrypto("BTCUSDT", Resolution.Minute, Market.Bybit).Symbol

self.universe = self.AddUniverse(CryptoUniverse.Bybit(self.UniverseSelectionFilter))
using QuantConnect.DataSource;
using QuantConnect.Data.UniverseSelection;
