The following snippet demonstrates how to request data from the Coinbase Crypto Price dataset:

from QuantConnect.DataSource import *
from QuantConnect.Data.UniverseSelection import *

# Coinbase only accepts Cash account type
self.SetBrokerageModel(BrokerageName.Coinbase, AccountType.Cash)

self.symbol = self.AddCrypto("BTCUSD", Resolution.Minute, Market.Coinbase).Symbol

self.universe = self.AddUniverse(CryptoUniverse.Coinbase(self.UniverseSelectionFilter))
using QuantConnect.DataSource;
using QuantConnect.Data.UniverseSelection;
