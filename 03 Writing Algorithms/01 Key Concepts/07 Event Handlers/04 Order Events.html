def OnOrderEvent(self, orderEvent: OrderEvent) -> None:
    order = self.Transactions.GetOrderById(orderEvent.OrderId)
    if orderEvent.Status == OrderStatus.Filled:
        self.Debug(f"{self.Time}: {order.Type}: {orderEvent}")

def OnAssignmentOrderEvent(self, assignmentEvent: OrderEvent) -> None:
    self.Log(str(assignmentEvent))

The OnOrderEvent event notifies the algorithm of new orders, and changes in the order status such as fill events and cancelations.

In backtesting order events are triggered synchronously after the main data events. In live trading, order events are asynchronously as order events occur. Market orders are fill synchronously
