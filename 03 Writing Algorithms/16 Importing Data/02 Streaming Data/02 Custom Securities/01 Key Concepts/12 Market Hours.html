The default market hours for custom securities is to be always open. To set the market hours, pass an exchangeHoursexchange_hours argument to the AddDataadd_data method. 

var symbol = QuantConnect.Symbol.Create("SPY", SecurityType.Equity, Market.USA);
var symbolProperties = SymbolPropertiesDatabase.GetSymbolProperties(symbol.ID.Market, symbol, SecurityType.Equity, "USD");
var securityExchangeHours = MarketHoursDatabase.GetExchangeHours(symbol.ID.Market, symbol, SecurityType.Equity);
var security = AddData<CustomType>(symbol, symbolProperties, securityExchangeHours, Resolution.Hour);
