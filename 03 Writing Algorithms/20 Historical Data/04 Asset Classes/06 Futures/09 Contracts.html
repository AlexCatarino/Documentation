To get the Symbol objects of historical contracts, call the future_chain_provider.get_future_contract_listFutureChainProvider.GetFutureContractList method with the continuous contract Symbol and a historical date. This method returns a list of Symbol objects that represent the contracts that were tradable between the historical date and the current day. 

public class FutureContractListHistoryAlgorithm : QCAlgorithm
{
    public override void Initialize()
    {
        SetStartDate(2024, 12, 19);
        // Add a Future.
        var future = AddFuture(Futures.Indices.SP500EMini);
        // Get the list of contracts that were tradable between one year ago and today.
        var contractSymbols = FutureChainProvider.GetFutureContractList(future.Symbol, Time.AddDays(-365));
    }
}
