To model zero transaction fees, use the ConstantFeeModel with zero as fee and specify the currency.

public override Initialize()
{
    var security = AddEquity("SPY");
    security.SetFeeModel(new ConstantFeeModel(fee: 0m, currency: "USD"));
}
def initialize(self):
    security = self.add_equity("SPY")
    security.set_fee_model(ConstantFeeModel(fee=0, currency="USD"))
