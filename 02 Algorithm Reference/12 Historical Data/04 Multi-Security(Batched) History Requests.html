Multi-security, or batched, history requests return data for a collection of secuity objects. They allow you to pass many assets into a single history call and get all of the data returned, pre-synchronized in a single collection. In Python, this data is returned as a Pandas Data Frame with multiple indexes. In CSharp, this data is returned as an enumerable of Slice objects. 

# Batch Request By Bars: 14 IBM, MSFT and AAPL TradeBars, return dataframe.

df = self.History(["IBM", "MSFT", "AAPL"], 14)
df = self.History(["IBM", "MSFT", "AAPL"], 14, Resolution.Minute)


# Batch Request By Period: 2 Weeks Of IBM Data, return dataframe.

df = self.History(["IBM", "MSFT", "AAPL"], timedelta(14))
df = self.History(["IBM", "MSFT", "AAPL"], timedelta(14), Resolution.Minute)


# Or, use batch API to fetch one security, but return dataframe.
df = self.History(["IBM"], 14)

// Batch Request By Bar Count: 14 IBM TradeBars 
var slices = History(new[] { "IBM", "MSFT", "AAPL" }, 14);
var slices = History(new[] { "IBM", "MSFT", "AAPL" }, 14, Resolution.Minute);

// Batch Request By Period: 2 Weeks Of IBM Data
var slices = History(new[] { "IBM", "MSFT", "AAPL" }, TimeSpan.FromDays(14));
var slices = History(new[] { "IBM", "MSFT", "AAPL" }, TimeSpan.FromDays(14), Resolution.Minute);
