You can create custom date rules with the FuncTimeRule class. The FuncTimeRule constructor expects a name argument of type stringstr and a createUtcEventTimesFunctioncreate_utc_event_times_function argument of type Func<IEnumerable<DateTime>, IEnumerable<DateTime>>Callable[List[datetime], List[datetime]].

var timeRule = new FuncTimeRule(
    name: "CustomTimeRule",
    createUtcEventTimesFunction: dates => dates.Select(d => d.AddHours(10)));

Schedule.On(DateRules.EveryDay(), timeRule, () => Debug(Time.ToString()));
def create_utc_event_times_function(dates: List[datetime]) -> List[datetime]:
    return [d + timedelta(hours=10) for d in dates]

time_rule = FuncTimeRule(
    name="CustomTimeRule", 
    create_utc_event_times_function=create_utc_event_times_function)

self.schedule.on(self.date_rules.every_day(), time_rule, lambda: self.debug(str(self.time)))
