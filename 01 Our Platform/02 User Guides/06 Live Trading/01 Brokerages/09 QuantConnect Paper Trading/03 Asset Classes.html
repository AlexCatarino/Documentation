The QuantConnect Paper Trading brokerage model supports trading Equities, Cryptocurrencies, Forex, and Futures.

// If SetBrokerageModel is correctly called, the market will be mapped automatically without explicitly stating it
AddCrypto("BTCUSDT", Resolution.Minute);
AddEquity("SPY", Resolution.Minute);
AddForex("EURUSD", Resolution.Minute);
AddFuture(Futures.Indices.SP500EMini, Resolution.Minute);
