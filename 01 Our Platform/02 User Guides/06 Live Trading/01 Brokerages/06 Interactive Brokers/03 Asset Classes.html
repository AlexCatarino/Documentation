The InteractiveBrokersBrokerageModel supports trading Equities, Options, Futures, and Future Options.

// If SetBrokerageModel is correctly called, the market will be mapped automatically without explicitly stating it
AddEquity("SPY", Resolution.Minute);
AddOption("SPY", Resolution.Minute);
AddFuture(Futures.Indices.SP500EMini, Resolution.Minute);

var goldFutures = AddFuture(Futures.Metals.Gold, Resolution.Minute);
goldFutures.SetFilter(0, 90);
AddFutureOption(goldFutures.Symbol, universe => universe.Strikes(-5, +5));
