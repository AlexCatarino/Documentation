The IB brokerage model supports trading Equities, Options, Futures, and Future Options.

AddEquities("SPY", Resolution.Minute);
AddOptions("SPY", Resolution.Minute);
AddFuture(Futures.Indices.SP500EMini, Resolution.Minute);
AddFutureOptions(Futures.Indices.SP500EMini, Resolution.Minute);
self.AddEquities("SPY", Resolution.Minute)
self.AddOptions("SPY", Resolution.Minute)
self.AddFuture(Futures.Indices.SP500EMini, Resolution.Minute)
self.AddFutureOptions(Futures.Indices.SP500EMini, Resolution.Minute)
